## 💻 代码

```cpp
#define LIGHT_SENSOR A0
#define LED_PIN 13

void setup() {
  pinMode(LED_PIN, OUTPUT);
  Serial.begin(9600);
}

void loop() {
  int lightValue = analogRead(LIGHT_SENSOR);
  Serial.print("光线值: ");
  Serial.println(lightValue);

  if (lightValue < 500) {
    digitalWrite(LED_PIN, HIGH);  // 暗 → 开灯
  } else {
    digitalWrite(LED_PIN, LOW);   // 亮 → 关灯
  }
  delay(200);
}
```
