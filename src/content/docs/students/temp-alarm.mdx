LED 红灯→D12（经220Ω），绿灯→D13（经220Ω），蜂鸣器→D8。

## 💻 代码

```cpp
#define TEMP_SENSOR A0
#define RED_LED 12
#define GREEN_LED 13
#define BUZZER 8

void setup() {
  pinMode(RED_LED, OUTPUT);
  pinMode(GREEN_LED, OUTPUT);
  pinMode(BUZZER, OUTPUT);
  Serial.begin(9600);
}

float readTemp() {
  int value = analogRead(TEMP_SENSOR);
  float voltage = value * (5.0 / 1023.0);
  return voltage * 100.0;  // LM35: 10mV/°C
}

void loop() {
  float temp = readTemp();
  Serial.print("温度: "); Serial.print(temp); Serial.println(" °C");

  if (temp > 30.0) {
    digitalWrite(RED_LED, HIGH);
    digitalWrite(GREEN_LED, LOW);
    tone(BUZZER, 1000, 200);  // 高温报警
  } else {
    digitalWrite(RED_LED, LOW);
    digitalWrite(GREEN_LED, HIGH);
  }
  delay(1000);
}
```
