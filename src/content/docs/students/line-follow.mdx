## 💻 代码

```cpp
#define LEFT_SENSOR 2
#define RIGHT_SENSOR 4
#define IN1 5
#define IN2 6
#define IN3 7
#define IN4 8
#define ENA 3
#define ENB 11

void setup() {
  pinMode(LEFT_SENSOR, INPUT);
  pinMode(RIGHT_SENSOR, INPUT);
  pinMode(IN1, OUTPUT); pinMode(IN2, OUTPUT);
  pinMode(IN3, OUTPUT); pinMode(IN4, OUTPUT);
  pinMode(ENA, OUTPUT); pinMode(ENB, OUTPUT);
}

void forward() {
  analogWrite(ENA, 120); analogWrite(ENB, 120);
  digitalWrite(IN1, HIGH); digitalWrite(IN2, LOW);
  digitalWrite(IN3, HIGH); digitalWrite(IN4, LOW);
}

void turnRight() {
  analogWrite(ENA, 100); analogWrite(ENB, 100);
  digitalWrite(IN1, HIGH); digitalWrite(IN2, LOW);
  digitalWrite(IN3, LOW); digitalWrite(IN4, HIGH);
}

void turnLeft() {
  analogWrite(ENA, 100); analogWrite(ENB, 100);
  digitalWrite(IN1, LOW); digitalWrite(IN2, HIGH);
  digitalWrite(IN3, HIGH); digitalWrite(IN4, LOW);
}

void loop() {
  int left = digitalRead(LEFT_SENSOR);   // 0 = 在黑线上
  int right = digitalRead(RIGHT_SENSOR);
  if (left == 0 && right == 0) { forward(); }
  else if (left == 1 && right == 0) { turnLeft(); }
  else if (left == 0 && right == 1) { turnRight(); }
  else { forward(); }
}
```
